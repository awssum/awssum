Loading AwsSum's Amazon SimpleDB

{% highlight javascript %} var awssum = require('awssum'); var amazon = awssum.load('amazon/amazon'); var SimpleDB = awssum.load('amazon/simpledb').SimpleDB; var sdb = new SimpleDB({ 'accessKeyId' : process.env.accessKeyId, 'secretAccessKey' : process.env.secretAccessKey, 'region' : amazon.US_EAST_1 }); {% endhighlight %}
