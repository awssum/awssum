name    : Get
title   : List instance metadata
intro   : This example shows how to list instance metadata. This can only be run inside an instance!
example : <<EOF
imd.Get(function(err, data) {
    //fails
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
imd.Get({},function(err, data) {
    //fails
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
var params = {
    Version  : 'latest',
    Category : '/'
};
imd.Get(params,function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
var params = {
    Version  : 'latest',
    Category : '/meta-data/'
};
imd.Get(params,function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
EOF
