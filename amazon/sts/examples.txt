title   : Get session token for 14 hours
intro   : This example shows how to get temporary security credentials that last 14 hours. Your temporary credentials can last between 1 and 36 hours and are default set to 12 hours.
example : <<EOF
var params = {
    DurationSeconds : 50400
}
sqs.GetSessionToken(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
EOF

name    : GetFederationToken
title   : Get federation token with a specific policy
intro   : This example shows how to get temporary security credentials with a specific policy. You will standard IAM policies and in this case it will allow all services on all resources. It will use the default duration (12 hours) while we omitted the DurationSeconds-property.
example : <<EOF
var params = {
    Name    : 'uniquename',
    Policy  :'{"Statement": [{"Effect": "Allow","Action": "*","Resource": "*"}]}'
}
sqs.GetFederationToken(params, function(err, data) {
    fmt.dump(err, 'Error');
    fmt.dump(data, 'Data');
});
EOF
